** Installation of my dotfiles
*Prerequisite* /rync/.

#+begin_src sh
cd $HOME
git clone --separate-git-dir=$HOME/Repos/dotfiles https://github.com/sadikeey/dotfiles.git tmpdotfiles

rsync --recursive --verbose --exclude '.git' tmpdotfiles/ $HOME/
rm -r tmpdotfiles
#+end_src
